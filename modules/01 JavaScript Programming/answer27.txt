function Validator(value)
{
    this.isBool=function(){
    return typeof(value)==="boolean";
    }
    this.isError=function(){
        return value instanceof Error;
    }
    this.isNaN=function(){
        return isNaN(value);
    }
    this.isNull=function(){
        return value===null;
    }
    this.isNumber=function(){
        return typeof(value)==="number";
    }
    this.isObject=function(){
        return typeof(value)==="Object";
    }
    this.isPureJson=function(){
        if (typeof value!=="string")
        return false;
        try {
            JSON.parse(value);//to check convertion to json
            return
        } catch (error) {
            return false;//failed to convert
        }
    }
    this.isRegExp = function(value){
        return (value instanceof RegExp)
    }
    this.isChar=function(value){
        return typeof(value)==="string"&&value.length===1;
    }
    this.isSameType=function(value2){
        return typeof(value)===typeof(value2);
    }
                                   
}

//---is boolean
var validate1=new Validator("1");
var validate2=new Validator(true);
console.log(validate1.isBool());
console.log(validate2.isBool());
//---is Error
var validate3=new Validator("1");
var validate4=new Validator(Error());
console.log(validate3.isError());
console.log(validate4.isError());

 var validate6=new Validator("1");
