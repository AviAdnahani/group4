var myFavRecipe = {
    title : "Mole",
    servings : 2,
    ingredients : ["cinnamon","cumin","cocoa"]
}

for(var key in myFavRecipe){
    switch (key){
        case "title" :
            console.log(myFavRecipe[key]);
            break;
        case "servings" :
            console.log("Serves: " + myFavRecipe[key]);
            break;
        case "ingredients" :
            console.log("Ingredients:")
            for(var i=0;i<myFavRecipe[key].length;i++)
                console.log(myFavRecipe[key][i]);
            break;
        default:
            break;
    }
}
